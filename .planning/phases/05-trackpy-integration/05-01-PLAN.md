The entrypoint MUST support BOTH modes for compatibility:

```python
def main():
    """Dual-mode entrypoint: legacy JSON or persistent NDJSON."""
    # Check if stdin has data immediately (legacy mode) or not (persistent mode)
    import select
    import sys
    
    # Try to detect mode: if stdin is a TTY or we're in persistent mode
    if sys.stdin.isatty():
        # Persistent worker mode: send ready handshake, then loop
        sys.stdout.write(json.dumps({"command": "ready", "version": TOOL_VERSION}) + "\n")
        sys.stdout.flush()
        _run_ndjson_loop()
    else:
        # Check if data is immediately available (legacy mode)
        # or if we should wait for handshake (persistent mode spawned with pipe)
        rlist, _, _ = select.select([sys.stdin], [], [], 0.1)
        if rlist:
            # Data available immediately - legacy single request mode
            line = sys.stdin.readline()
            if line.strip():
                request = json.loads(line)
                response = _handle_request(request)
                sys.stdout.write(json.dumps(response) + "\n")
                sys.stdout.flush()
                return
        
        # No immediate data - persistent worker mode
        sys.stdout.write(json.dumps({"command": "ready", "version": TOOL_VERSION}) + "\n")
        sys.stdout.flush()
        _run_ndjson_loop()
```

```python
def handle_meta_list(params: dict) -> dict:
    """Out-of-process function discovery.
    
    Called by core server via execute_tool() to discover trackpy functions
    without importing trackpy in the server process.
    """
    import trackpy
    from numpydoc.docscrape import NumpyDocString
    
    functions = []
    for module_name in TRACKPY_MODULES:
        try:
            module = importlib.import_module(module_name)
        except ImportError:
            continue
        
        for name in dir(module):
            if name.startswith('_'):
                continue
            obj = getattr(module, name)
            if not callable(obj) or not hasattr(obj, '__doc__'):
                continue
            
            fn_id = f"{module_name}.{name}"
            doc = obj.__doc__ or ""
            summary = doc.split('\n')[0] if doc else ""
            
            functions.append({
                "fn_id": fn_id,
                "name": name,
                "summary": summary,
                "module": module_name,
            })
    
    return {"ok": True, "result": {"functions": functions}}
```
