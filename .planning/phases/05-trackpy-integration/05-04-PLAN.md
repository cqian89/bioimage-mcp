import pytest
from unittest.mock import MagicMock, patch
from bioimage_mcp.runtimes.persistent import WorkerProcess

def test_worker_kills_process_on_read_error():
    """Verify worker process is killed when reading response fails."""
    # Mock process
    mock_proc = MagicMock()
    mock_proc.poll.return_value = None  # Running
    mock_proc.stdout.readline.side_effect = RuntimeError("Connection lost")
    
    # Create worker with mocked process
    worker = WorkerProcess("test-env", ["python"], "/tmp")
    worker._process = mock_proc
    worker._state = "READY"
    
    # Execute should catch error, kill process, and re-raise
    with pytest.raises(RuntimeError, match="Connection lost"):
        worker.execute({"test": "cmd"})
            
    # Assert kill was called
    mock_proc.kill.assert_called_once()
    assert worker._state == "TERMINATED"
