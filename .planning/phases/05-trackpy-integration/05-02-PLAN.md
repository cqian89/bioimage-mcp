```python
"""Introspection helpers for trackpy functions.

Runs inside the trackpy env to support out-of-process discovery.
"""
from __future__ import annotations

import importlib
import inspect
from typing import Any

from numpydoc.docscrape import NumpyDocString

TRACKPY_MODULES = [
    "trackpy",
    "trackpy.linking",
    "trackpy.motion",
    # ... full list
]

# Params that are artifact inputs, not scalar params
ARTIFACT_INPUT_PARAMS = {
    "image", "raw_image", "frames", "f", 
    "features", "t", "tracks", "traj",
    "label_image", "mask",
}


def introspect_module(module_name: str) -> list[dict]:
    """Discover all public functions in a module."""
    try:
        module = importlib.import_module(module_name)
    except ImportError:
        return []
    
    functions = []
    for name in dir(module):
        if name.startswith('_'):
            continue
        obj = getattr(module, name)
        if not callable(obj) or inspect.isclass(obj):
            continue
        
        fn_id = f"{module_name}.{name}"
        summary = _extract_summary(obj)
        io_pattern = _determine_io_pattern(module_name, name)
        
        functions.append({
            "fn_id": fn_id,
            "name": name,
            "summary": summary,
            "module": module_name,
            "io_pattern": io_pattern,
        })
    
    return functions


def introspect_function(fn_id: str) -> dict:
    """Get detailed schema for a single function."""
    parts = fn_id.rsplit(".", 1)
    module_name, func_name = parts if len(parts) == 2 else ("trackpy", parts[0])
    
    module = importlib.import_module(module_name)
    func = getattr(module, func_name)
    
    sig = inspect.signature(func)
    doc = NumpyDocString(func.__doc__ or "")
    
    # Build params schema
    properties = {}
    required = []
    
    for param_name, param in sig.parameters.items():
        if param_name in ARTIFACT_INPUT_PARAMS:
            continue  # Filter artifact inputs
        if param.kind == inspect.Parameter.VAR_KEYWORD:
            continue  # Skip **kwargs
        
        prop = _param_to_schema(param, doc)
        properties[param_name] = prop
        
        if param.default is inspect.Parameter.empty:
            required.append(param_name)
    
    return {
        "fn_id": fn_id,
        "params_schema": {
            "type": "object",
            "properties": properties,
            "required": required,
        },
        "summary": _extract_summary(func),
        "introspection_source": "numpydoc",
    }


def _extract_summary(func) -> str:
    doc = func.__doc__ or ""
    return doc.split('\n')[0].strip()


def _determine_io_pattern(module_name: str, func_name: str) -> str:
    # locate, batch, refine -> IMAGE_TO_TABLE (features DataFrame)
    # link, link_df -> TABLE_TO_TABLE (trajectories DataFrame)
    # msd, imsd, emsd -> TABLE_TO_TABLE (statistics DataFrame)
    # annotate, plot_traj -> IMAGE_TO_IMAGE (annotated image)
    if func_name in ("locate", "batch", "refine", "local_maxima"):
        return "IMAGE_TO_TABLE"
    if func_name in ("link", "link_df", "link_iter"):
        return "TABLE_TO_TABLE"
    if func_name in ("msd", "imsd", "emsd", "vanhove"):
        return "TABLE_TO_TABLE"
    if func_name in ("annotate", "plot_traj", "annotate3d"):
        return "IMAGE_TO_IMAGE"
    return "GENERIC"
```

```python
def handle_meta_list(params: dict) -> dict:
    """Out-of-process function discovery.
    
    Called by core server via execute_tool() to discover trackpy functions
    without importing trackpy in the server process.
    """
    functions = []
    for module_name in TRACKPY_MODULES:
        module_funcs = introspect_module(module_name)
        functions.extend(module_funcs)
    
    return {"ok": True, "result": {"functions": functions}}


def handle_meta_describe(params: dict) -> dict:
    """Out-of-process function schema retrieval."""
    target_fn = params.get("target_fn")
    if not target_fn:
        return {"ok": False, "error": {"message": "target_fn required"}}
    
    result = introspect_function(target_fn)
    return {"ok": True, "result": result}


# In main loop, add handlers:
```

```python
from bioimage_mcp.runtimes.executor import execute_tool


def _discover_via_subprocess(manifest: ToolManifest) -> list[FunctionMetadata]:
    """Discover functions via out-of-process meta.list call.
    
    Used when tool env has different Python version or dependencies
    that cannot be imported in the server process.
    """
    entrypoint = manifest.entrypoint
    entry_path = Path(entrypoint)
    if not entry_path.is_absolute():
        candidate = manifest.manifest_path.parent / entry_path
        if candidate.exists():
            entrypoint = str(candidate)
    
    request = {
        "fn_id": "meta.list",
        "params": {},
        "inputs": {},
    }
    
    response, log_text, exit_code = execute_tool(
        entrypoint=entrypoint,
        request=request,
        env_id=manifest.env_id,
        timeout_seconds=60,
    )
    
    if not response.get("ok"):
        logger.warning("meta.list failed for %s: %s", manifest.tool_id, response)
        return []
    
    result = response.get("result", {})
    functions = result.get("functions", [])
    
    return [
        FunctionMetadata(
            fn_id=f["fn_id"],
            name=f["name"],
            summary=f.get("summary", ""),
            module=f.get("module", ""),
            io_pattern=f.get("io_pattern", "GENERIC"),
            source_adapter="subprocess",
        )
        for f in functions
    ]
```

```python
"""API coverage report generator for trackpy integration.

Produces TRACK-03 compliance artifact documenting which functions are
exposed vs excluded and why.
"""
from __future__ import annotations

import importlib
import inspect
from dataclasses import dataclass, field
from typing import Any

from bioimage_mcp_trackpy.introspect import TRACKPY_MODULES


@dataclass
class ExcludedFunction:
    """A function that was excluded from exposure."""
    fn_id: str
    name: str
    module: str
    reason: str
    details: str = ""


@dataclass  
class CoverageReport:
    """API coverage report for a module set."""
    total_scanned: int = 0
    total_exposed: int = 0
    excluded: list[ExcludedFunction] = field(default_factory=list)
    by_module: dict[str, dict[str, int]] = field(default_factory=dict)
    
    @property
    def coverage_pct(self) -> float:
        if self.total_scanned == 0:
            return 0.0
        return (self.total_exposed / self.total_scanned) * 100


EXCLUSION_REASONS = {
    "private": "Name starts with underscore",
    "class": "Is a class, not a function",
    "no_doc": "Missing docstring (undocumented)",
    "kwargs_only": "Only accepts **kwargs (no introspectable params)",
    "deprecated": "Marked as deprecated",
    "alias": "Alias/re-export of another function",
    "internal": "Internal implementation detail",
}


def _is_excluded(name: str, obj: Any, module_name: str) -> tuple[bool, str, str]:
    """Check if a callable should be excluded. Returns (excluded, reason_key, details)."""
    
    # Private names
    if name.startswith('_'):
        return True, "private", f"Name '{name}' starts with underscore"
    
    # Classes (we expose functions, not classes directly)
    if inspect.isclass(obj):
        return True, "class", f"{name} is a class"
    
    # Not callable
    if not callable(obj):
        return True, "internal", f"{name} is not callable"
    
    # No docstring
    if not obj.__doc__:
        return True, "no_doc", f"{name} has no docstring"
    
    # Check for deprecated marker in docstring
    doc = obj.__doc__ or ""
    if "deprecated" in doc.lower()[:200]:
        return True, "deprecated", f"{name} marked deprecated in docstring"
    
    # Check signature for kwargs-only
    try:
        sig = inspect.signature(obj)
        params = list(sig.parameters.values())
        
        # If ONLY has **kwargs (no other params)
        if len(params) == 1 and params[0].kind == inspect.Parameter.VAR_KEYWORD:
            return True, "kwargs_only", f"{name} only accepts **kwargs"
    except (ValueError, TypeError):
        pass  # Can't get signature, don't exclude for this reason
    
    # Check if it's a re-export from another module
    if hasattr(obj, '__module__') and obj.__module__ != module_name:
        # Only exclude if it's from a submodule being scanned separately
        if obj.__module__ in TRACKPY_MODULES:
            return True, "alias", f"{name} is re-exported from {obj.__module__}"
    
    return False, "", ""


def generate_coverage_report() -> CoverageReport:
    """Generate full API coverage report for trackpy modules."""
    report = CoverageReport()
    
    for module_name in TRACKPY_MODULES:
        try:
            module = importlib.import_module(module_name)
        except ImportError:
            continue
        
        module_stats = {"scanned": 0, "exposed": 0, "excluded": 0}
        
        for name in dir(module):
            obj = getattr(module, name, None)
            if obj is None:
                continue
            
            # Only count callables
            if not callable(obj) and not inspect.isclass(obj):
                continue
                
            module_stats["scanned"] += 1
            report.total_scanned += 1
            
            excluded, reason_key, details = _is_excluded(name, obj, module_name)
            
            if excluded:
                module_stats["excluded"] += 1
                report.excluded.append(ExcludedFunction(
                    fn_id=f"{module_name}.{name}",
                    name=name,
                    module=module_name,
                    reason=EXCLUSION_REASONS.get(reason_key, reason_key),
                    details=details,
                ))
            else:
                module_stats["exposed"] += 1
                report.total_exposed += 1
        
        report.by_module[module_name] = module_stats
    
    return report


def format_report_markdown(report: CoverageReport) -> str:
    """Format coverage report as Markdown."""
    lines = [
        "# Trackpy API Coverage Report (TRACK-03)",
        "",
        "## Summary",
        "",
        f"- **Total scanned:** {report.total_scanned}",
        f"- **Total exposed:** {report.total_exposed}",
        f"- **Total excluded:** {len(report.excluded)}",
        f"- **Coverage:** {report.coverage_pct:.1f}%",
        "",
        "## Coverage by Module",
        "",
        "| Module | Scanned | Exposed | Excluded |",
        "|--------|---------|---------|----------|",
    ]
    
    for mod, stats in sorted(report.by_module.items()):
        lines.append(f"| {mod} | {stats['scanned']} | {stats['exposed']} | {stats['excluded']} |")
    
    lines.extend([
        "",
        "## Exclusions",
        "",
        "| Function | Reason | Details |",
        "|----------|--------|---------|",
    ])
    
    for exc in sorted(report.excluded, key=lambda x: x.fn_id):
        lines.append(f"| `{exc.fn_id}` | {exc.reason} | {exc.details} |")
    
    return "\n".join(lines)


if __name__ == "__main__":
    # CLI for generating report
    report = generate_coverage_report()
    print(format_report_markdown(report))
```
