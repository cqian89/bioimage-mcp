"""Equivalence smoke test for trackpy integration.

Compares MCP trackpy execution against native trackpy baseline.

Run with: pytest -m "smoke_full and requires_env" tests/smoke/test_equivalence_trackpy.py -v

@pytest.mark.smoke_full
@pytest.mark.requires_env("bioimage-mcp-trackpy")
@pytest.mark.anyio
async def test_trackpy_locate_equivalence(
    live_server, trackpy_example_frame, helper, native_executor, tmp_path
):
    """Test that MCP trackpy.locate matches native trackpy.locate.
    
    Validates:
    - Feature count within ±20% tolerance
    - Mean position coordinates within tolerance
    - Mass values within tolerance
    """
    diameter = 7
    minmass = 0.1
    
    # 1. Run MCP trackpy.locate
    # First load image via MCP
    load_result = await live_server.call_tool(
        "run",
        {
            "fn_id": "base.io.bioimage.load",
            "inputs": {},
            "params": {"path": str(trackpy_example_frame)},
        },
    )
    assert load_result["status"] == "success", f"Load failed: {load_result.get('error')}"
    img_ref = load_result["outputs"]["image"]
    
    # Run locate via MCP
    mcp_result = await live_server.call_tool(
        "run",
        {
            "fn_id": "trackpy.locate",
            "inputs": {"image": img_ref},
            "params": {"diameter": diameter, "minmass": minmass},
        },
    )
    assert mcp_result.get("status") == "success", f"MCP locate failed: {mcp_result}"
    mcp_output_ref = mcp_result["outputs"]["table"]  # TableRef
    
    # 2. Run Native Baseline
    baseline_script = Path(__file__).parent / "reference_scripts" / "trackpy_baseline.py"
    native_output_path = tmp_path / "native_features.csv"
    
    baseline_result = native_executor.run_script(
        env_name="bioimage-mcp-trackpy",
        script_path=baseline_script,
        args=[
            "--input", str(trackpy_example_frame),
            "--output", str(native_output_path),
            "--diameter", str(diameter),
            "--minmass", str(minmass),
            "--mode", "locate",
        ],
    )
    assert baseline_result["status"] == "success", f"Baseline failed: {baseline_result}"
    
    # 3. Load outputs and compare
    # MCP output must be cross-env portable (file-backed TableRef).
    mcp_uri = mcp_output_ref.get("uri", "")
    assert mcp_uri.startswith("file:"), f"Expected file-backed TableRef, got {mcp_uri}"
    mcp_path = Path(mcp_uri.replace("file://", "").replace("file:", ""))

    # Load both feature tables
    mcp_features = pd.read_csv(mcp_path)
    native_features = pd.read_csv(native_output_path)

    # Normalize ordering/columns for stable comparisons
    compare_cols = [c for c in ["x", "y", "mass"] if c in mcp_features.columns and c in native_features.columns]
    assert compare_cols, f"No comparable columns between MCP and native: {mcp_features.columns} vs {native_features.columns}"
    mcp_features = mcp_features[compare_cols].sort_values(compare_cols).reset_index(drop=True)
    native_features = native_features[compare_cols].sort_values(compare_cols).reset_index(drop=True)
    
    # 4. Tolerance-based comparison
    # Feature count: within ±20%
    mcp_count = len(mcp_features)
    native_count = len(native_features)
    
    assert mcp_count > 0, "MCP found no features"
    assert native_count > 0, "Native found no features"
    
    count_ratio = mcp_count / native_count
    assert 0.8 <= count_ratio <= 1.2, (
        f"Feature count mismatch: MCP={mcp_count}, native={native_count}, ratio={count_ratio:.2f}"
    )
    
    # Use helper for table-level equivalence on comparable columns (tolerant float compare)
    helper.assert_table_equivalent(mcp_features, native_features, rtol=1e-2)

"""Integration smoke tests for trackpy.

These tests validate that trackpy functions execute correctly.
Run INSIDE the trackpy env:

from __future__ import annotations

import tempfile
from pathlib import Path

import numpy as np
import pytest

# Skip entire module if trackpy not installed
tp = pytest.importorskip("trackpy")
import pandas as pd


@pytest.fixture
def synthetic_2d_frame():
    """Generate a single 2D frame with synthetic particles.
    
    Creates a 256x256 image with ~10 Gaussian particles at known positions.
    Uses trackpy.artificial.draw_spots which is deterministic given seed.
    """
    import trackpy.artificial as ta
    
    np.random.seed(42)  # Deterministic for reproducibility
    
    # Generate random positions for particles
    n_particles = 10
    positions = np.random.rand(n_particles, 2) * 200 + 28  # Keep away from edges
    
    # Create blank image and draw spots
    frame = np.zeros((256, 256), dtype=np.float64)
    frame = ta.draw_spots(frame, positions, size=3.0, noise=0.1)
    
    return {
        "image": frame,
        "expected_positions": positions,
        "expected_count": n_particles,
    }


@pytest.fixture
def synthetic_2d_movie():
    """Generate a 2D movie (10 frames) with moving particles.
    
    Creates particles that move with constant velocity + noise.
    Useful for testing link() function.
    """
    import trackpy.artificial as ta
    
    np.random.seed(42)
    
    n_frames = 10
    n_particles = 5
    size = 256
    
    # Initial positions
    initial_positions = np.random.rand(n_particles, 2) * 150 + 50
    
    # Velocities (pixels per frame)
    velocities = np.random.randn(n_particles, 2) * 2
    
    frames = []
    
    for t in range(n_frames):
        # Update positions
        positions = initial_positions + velocities * t
        
        # Add small noise
        positions += np.random.randn(n_particles, 2) * 0.5
        
        # Create frame
        frame = np.zeros((size, size), dtype=np.float64)
        # Filter positions that are in bounds
        valid_mask = np.all((positions > 10) & (positions < size - 10), axis=1)
        valid_positions = positions[valid_mask]
        
        frame = ta.draw_spots(frame, valid_positions, size=3.0, noise=0.1)
        frames.append(frame)
    
    return {
        "frames": np.stack(frames),  # Shape: (10, 256, 256)
        "n_frames": n_frames,
        "n_particles": n_particles,
    }


@pytest.mark.requires_env("bioimage-mcp-trackpy")
class TestTrackpyLocate:
    """Tests for trackpy.locate feature finding."""
    
    def test_locate_finds_particles_in_2d(self, synthetic_2d_frame):
        """Locate should find approximately the expected number of particles."""
        image = synthetic_2d_frame["image"]
        expected_count = synthetic_2d_frame["expected_count"]
        
        features = tp.locate(image, diameter=7, minmass=0.1)
        
        # Tolerance-based validation
        found_count = len(features)
        assert found_count >= expected_count * 0.7, \
            f"Expected ~{expected_count} particles, found {found_count}"
        assert found_count <= expected_count * 1.3, \
            f"Found too many particles: {found_count} vs expected ~{expected_count}"
        
        # Check features have required columns
        assert "x" in features.columns
        assert "y" in features.columns
        assert "mass" in features.columns
    
    def test_locate_returns_dataframe(self, synthetic_2d_frame):
        """Locate should return a pandas DataFrame."""
        image = synthetic_2d_frame["image"]
        features = tp.locate(image, diameter=7, minmass=0.1)
        
        assert isinstance(features, pd.DataFrame)
        assert len(features) > 0


@pytest.mark.requires_env("bioimage-mcp-trackpy")
class TestTrackpyLink:
    """Tests for trackpy.link trajectory linking."""
    
    def test_link_creates_particle_column(self, synthetic_2d_movie):
        """Link should add particle column to features DataFrame."""
        frames = synthetic_2d_movie["frames"]
        
        # Locate features in all frames
        all_features = []
        for i, frame in enumerate(frames):
            features = tp.locate(frame, diameter=7, minmass=0.1)
            features["frame"] = i
            all_features.append(features)
        
        if not all_features or all(len(f) == 0 for f in all_features):
            pytest.skip("No features found in movie")
        
        features_df = pd.concat(all_features, ignore_index=True)
        
        # Link trajectories
        trajectories = tp.link(features_df, search_range=10, memory=2)
        
        assert "particle" in trajectories.columns
        assert trajectories["particle"].nunique() > 0
    
    def test_link_tracks_particles_across_frames(self, synthetic_2d_movie):
        """Link should create trajectories spanning multiple frames."""
        frames = synthetic_2d_movie["frames"]
        
        # Locate and link
        all_features = []
        for i, frame in enumerate(frames):
            features = tp.locate(frame, diameter=7, minmass=0.1)
            features["frame"] = i
            all_features.append(features)
        
        if not all_features or all(len(f) == 0 for f in all_features):
            pytest.skip("No features found in movie")
        
        features_df = pd.concat(all_features, ignore_index=True)
        trajectories = tp.link(features_df, search_range=10, memory=2)
        
        # Check some trajectories span multiple frames
        traj_lengths = trajectories.groupby("particle")["frame"].count()
        max_length = traj_lengths.max()
        
        # At least one trajectory should span multiple frames
        assert max_length >= 2, "No trajectories span multiple frames"

@pytest.mark.requires_env("bioimage-mcp-trackpy")
class TestTrackpyBatch:
    """Tests for trackpy.batch (UAT gap closure)."""

    def test_batch_processes_movie_stack(self, synthetic_2d_movie):
        """Batch should process in-memory numpy stack (frames, y, x)."""
        frames = synthetic_2d_movie["frames"]
        
        # Run batch directly on the numpy stack
        features = tp.batch(frames, diameter=7, minmass=0.1)
        
        assert isinstance(features, pd.DataFrame)
        assert len(features) > 0
        assert "frame" in features.columns
        # Should have features from multiple frames
        assert features["frame"].nunique() > 1
