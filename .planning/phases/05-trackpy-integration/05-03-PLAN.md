```python
"""Native trackpy baseline for equivalence testing.

Runs trackpy.locate directly (outside MCP) to provide ground truth.
Output is compared against MCP execution with tolerance-based checks.

Usage:
    python trackpy_baseline.py --input image.ome.tiff --output features.csv \
        --diameter 7 --minmass 0.1
"""
from __future__ import annotations

import argparse
import json
import os
import random
from pathlib import Path

import numpy as np
import pandas as pd
from bioio import BioImage


def set_seeds(seed: int = 42):
    """Set random seeds for reproducibility."""
    random.seed(seed)
    os.environ["PYTHONHASHSEED"] = str(seed)
    np.random.seed(seed)


def main():
    parser = argparse.ArgumentParser(description="Run trackpy baseline.")
    parser.add_argument("--input", required=True, help="Path to input image")
    parser.add_argument("--output", required=True, help="Path to save features CSV")
    parser.add_argument("--diameter", type=int, default=7, help="Feature diameter (odd int)")
    parser.add_argument("--minmass", type=float, default=0.1, help="Minimum mass threshold")
    parser.add_argument("--seed", type=int, default=42, help="Random seed")
    parser.add_argument("--mode", choices=["locate", "link"], default="locate", 
                        help="Which function to run")
    
    args = parser.parse_args()
    set_seeds(args.seed)
    
    import trackpy as tp
    
    input_path = Path(args.input)
    output_path = Path(args.output)
    output_path.parent.mkdir(parents=True, exist_ok=True)
    
    # Load image
    img = BioImage(input_path)
    data = img.data.squeeze()
    if hasattr(data, "values"):
        data = data.values
    
    if args.mode == "locate":
        # Run locate
        features = tp.locate(data, diameter=args.diameter, minmass=args.minmass)
        
        # Save as CSV
        features.to_csv(output_path, index=False)
        
        # JSON output contract
        print(json.dumps({
            "status": "success",
            "mode": "locate",
            "output_path": str(output_path),
            "n_features": len(features),
            "columns": list(features.columns),
            "mean_mass": float(features["mass"].mean()) if len(features) > 0 else 0.0,
        }))
    
    elif args.mode == "link":
        # For link mode, expect input to be a movie (3D: T, Y, X)
        # First locate in each frame, then link
        all_features = []
        for t in range(data.shape[0]):
            frame = data[t]
            features = tp.locate(frame, diameter=args.diameter, minmass=args.minmass)
            features["frame"] = t
            all_features.append(features)
        
        if not all_features or all(len(f) == 0 for f in all_features):
            print(json.dumps({
                "status": "success",
                "mode": "link",
                "output_path": str(output_path),
                "n_trajectories": 0,
                "warning": "No features found in any frame",
            }))
            return
        
        features_df = pd.concat(all_features, ignore_index=True)
        trajectories = tp.link(features_df, search_range=10, memory=2)
        
        # Save as CSV
        trajectories.to_csv(output_path, index=False)
        
        print(json.dumps({
            "status": "success",
            "mode": "link",
            "output_path": str(output_path),
            "n_trajectories": int(trajectories["particle"].nunique()),
            "n_features": len(trajectories),
        }))


if __name__ == "__main__":
    main()
```

    """Generate a small synthetic image with particle-like spots.
    
    Uses trackpy.artificial for reproducible particle generation.
    Runs in a subprocess in trackpy env to avoid importing trackpy in server env.
    """
    import subprocess
    import json
    
    # Use datasets/synthetic which is explicitly allowed in config
    allowed_tmp = Path.cwd() / "datasets" / "synthetic"
    allowed_tmp.mkdir(parents=True, exist_ok=True)
    
    import uuid
    image_path = allowed_tmp / f"synthetic_particles_{uuid.uuid4().hex[:8]}.ome.tiff"
    
    # Generate image via subprocess in trackpy env
    gen_script = f'''
import numpy as np
import json
np.random.seed(42)

try:
    import trackpy.artificial as ta
    
    # Generate random positions for particles
    n_particles = 10
    positions = np.random.rand(n_particles, 2) * 200 + 28  # Keep away from edges
    
    # Create blank image and draw spots
    frame = np.zeros((256, 256), dtype=np.float64)
    frame = ta.draw_spots(frame, positions, size=3.0, noise=0.1)
    
    # Save as OME-TIFF
    from bioio.writers import OmeTiffWriter
    OmeTiffWriter.save(frame.astype(np.float32), "{image_path}", dim_order="YX")
    
    print(json.dumps({{"status": "success", "n_particles": n_particles, "path": "{image_path}"}}))
except Exception as e:
    print(json.dumps({{"status": "error", "error": str(e)}}))
'''
    
    result = subprocess.run(
        ["micromamba", "run", "-n", "bioimage-mcp-trackpy", "python", "-c", gen_script],
        capture_output=True,
        text=True,
        timeout=60,
    )
    
    if result.returncode != 0:
        pytest.skip(f"Failed to generate synthetic image: {result.stderr}")
    
    try:
        output = json.loads(result.stdout.strip().split('\n')[-1])
        if output.get("status") != "success":
            pytest.skip(f"Synthetic generation failed: {output.get('error')}")
    except json.JSONDecodeError:
        pytest.skip(f"Failed to parse generation output: {result.stdout}")
    
    yield image_path
    
    # Cleanup
    if image_path.exists():
        image_path.unlink()


@pytest.mark.smoke_full
@pytest.mark.requires_env("bioimage-mcp-trackpy")
@pytest.mark.anyio
async def test_trackpy_locate_equivalence(
    live_server, synthetic_particle_image, helper, native_executor, tmp_path
):
    """Test that MCP trackpy.locate matches native trackpy.locate.
    
    Validates:
    - Feature count within ±20% tolerance
    - Mean position coordinates within tolerance
    - Mass values within tolerance
    """
    diameter = 7
    minmass = 0.1
    
    # 1. Run MCP trackpy.locate
    # First load image via MCP
    load_result = await live_server.call_tool(
        "run",
        {
            "fn_id": "base.io.bioimage.load",
            "inputs": {},
            "params": {"path": str(synthetic_particle_image)},
        },
    )
    assert load_result["status"] == "success", f"Load failed: {load_result.get('error')}"
    img_ref = load_result["outputs"]["image"]
    
    # Run locate via MCP
    mcp_result = await live_server.call_tool(
        "run",
        {
            "fn_id": "trackpy.locate",
            "inputs": {"image": img_ref},
            "params": {"diameter": diameter, "minmass": minmass},
        },
    )
    assert mcp_result.get("status") == "success", f"MCP locate failed: {mcp_result}"
    mcp_output_ref = mcp_result["outputs"]["features"]  # TableRef
    
    # 2. Run Native Baseline
    baseline_script = Path(__file__).parent / "reference_scripts" / "trackpy_baseline.py"
    native_output_path = tmp_path / "native_features.csv"
    
    baseline_result = native_executor.run_script(
        env_name="bioimage-mcp-trackpy",
        script_path=baseline_script,
        args=[
            "--input", str(synthetic_particle_image),
            "--output", str(native_output_path),
            "--diameter", str(diameter),
            "--minmass", str(minmass),
            "--mode", "locate",
        ],
    )
    assert baseline_result["status"] == "success", f"Baseline failed: {baseline_result}"
    
    # 3. Load outputs and compare
    # MCP output: get URI from ref
    mcp_uri = mcp_output_ref.get("uri", "")
    if mcp_uri.startswith("file://"):
        mcp_path = Path(mcp_uri.replace("file://", ""))
    else:
        # Handle in-memory ref - may need to export
        pytest.skip("MCP returned in-memory ref, export not implemented")
    
    # Load both feature tables
    mcp_features = pd.read_csv(mcp_path) if mcp_path.suffix == ".csv" else pd.read_parquet(mcp_path)
    native_features = pd.read_csv(native_output_path)
    
    # 4. Tolerance-based comparison
    # Feature count: within ±20%
    mcp_count = len(mcp_features)
    native_count = len(native_features)
    
    assert mcp_count > 0, "MCP found no features"
    assert native_count > 0, "Native found no features"
    
    count_ratio = mcp_count / native_count
    assert 0.8 <= count_ratio <= 1.2, (
        f"Feature count mismatch: MCP={mcp_count}, native={native_count}, ratio={count_ratio:.2f}"
    )
    
    # Mean mass: within ±30% (mass values can vary with implementation)
    if "mass" in mcp_features.columns and "mass" in native_features.columns:
        mcp_mean_mass = mcp_features["mass"].mean()
        native_mean_mass = native_features["mass"].mean()
        
        if native_mean_mass > 0:
            mass_ratio = mcp_mean_mass / native_mean_mass
            assert 0.7 <= mass_ratio <= 1.3, (
                f"Mass mismatch: MCP mean={mcp_mean_mass:.2f}, native mean={native_mean_mass:.2f}"
            )

"""Integration smoke tests for trackpy.

These tests validate that trackpy functions execute correctly.
Run INSIDE the trackpy env:

import pytest

# Skip entire module if trackpy not installed

@pytest.fixture
def synthetic_2d_frame():

    Creates a 256x256 image with ~10 Gaussian particles at known positions.
    Uses trackpy.artificial.draw_spots which is deterministic given seed.

@pytest.fixture
def synthetic_2d_movie():
    """Generate a 2D movie (10 frames) with moving particles.
    
    Creates particles that move with constant velocity + noise.
    Useful for testing link() function.
    """
    import trackpy.artificial as ta
    
    np.random.seed(42)
    
    n_frames = 10
    n_particles = 5
    size = 256
    
    # Initial positions
    initial_positions = np.random.rand(n_particles, 2) * 150 + 50
    
    # Velocities (pixels per frame)
    velocities = np.random.randn(n_particles, 2) * 2
    
    frames = []
    
    for t in range(n_frames):
        # Update positions
        positions = initial_positions + velocities * t
        
        # Add small noise
        positions += np.random.randn(n_particles, 2) * 0.5
        
        # Create frame
        frame = np.zeros((size, size), dtype=np.float64)
        # Filter positions that are in bounds
        valid_mask = np.all((positions > 10) & (positions < size - 10), axis=1)
        valid_positions = positions[valid_mask]
        
        frame = ta.draw_spots(frame, valid_positions, size=3.0, noise=0.1)
        frames.append(frame)
    
    return {
        "frames": np.stack(frames),  # Shape: (10, 256, 256)
        "n_frames": n_frames,
        "n_particles": n_particles,
    }


@pytest.mark.requires_env("bioimage-mcp-trackpy")
class TestTrackpyLocate:
    """Tests for trackpy.locate feature finding."""
    
    def test_locate_finds_particles_in_2d(self, synthetic_2d_frame):
        """Locate should find approximately the expected number of particles."""
        image = synthetic_2d_frame["image"]
        expected_count = synthetic_2d_frame["expected_count"]
        
        features = tp.locate(image, diameter=7, minmass=0.1)
        
        # Tolerance-based validation
        found_count = len(features)
        assert found_count >= expected_count * 0.7, \
            f"Expected ~{expected_count} particles, found {found_count}"
        assert found_count <= expected_count * 1.3, \
            f"Found too many particles: {found_count} vs expected ~{expected_count}"
        
        # Check features have required columns
        assert "x" in features.columns
        assert "y" in features.columns
        assert "mass" in features.columns
    
    def test_locate_returns_dataframe(self, synthetic_2d_frame):
        """Locate should return a pandas DataFrame."""
        image = synthetic_2d_frame["image"]
        features = tp.locate(image, diameter=7, minmass=0.1)
        
        assert isinstance(features, pd.DataFrame)
        assert len(features) > 0


@pytest.mark.requires_env("bioimage-mcp-trackpy")
class TestTrackpyLink:
    """Tests for trackpy.link trajectory linking."""
    
    def test_link_creates_particle_column(self, synthetic_2d_movie):
        """Link should add particle column to features DataFrame."""
        frames = synthetic_2d_movie["frames"]
        
        # Locate features in all frames
        all_features = []
        for i, frame in enumerate(frames):
            features = tp.locate(frame, diameter=7, minmass=0.1)
            features["frame"] = i
            all_features.append(features)
        
        if not all_features or all(len(f) == 0 for f in all_features):
            pytest.skip("No features found in movie")
        
        features_df = pd.concat(all_features, ignore_index=True)
        
        # Link trajectories
        trajectories = tp.link(features_df, search_range=10, memory=2)
        
        assert "particle" in trajectories.columns
        assert trajectories["particle"].nunique() > 0
    
    def test_link_tracks_particles_across_frames(self, synthetic_2d_movie):
        """Link should create trajectories spanning multiple frames."""
        frames = synthetic_2d_movie["frames"]
        
        # Locate and link
        all_features = []
        for i, frame in enumerate(frames):
            features = tp.locate(frame, diameter=7, minmass=0.1)
            features["frame"] = i
            all_features.append(features)
        
        if not all_features or all(len(f) == 0 for f in all_features):
            pytest.skip("No features found in movie")
        
        features_df = pd.concat(all_features, ignore_index=True)
        trajectories = tp.link(features_df, search_range=10, memory=2)
        
        # Check some trajectories span multiple frames
        traj_lengths = trajectories.groupby("particle")["frame"].count()
        max_length = traj_lengths.max()
        
        # At least one trajectory should span multiple frames
        assert max_length >= 2, "No trajectories span multiple frames"
