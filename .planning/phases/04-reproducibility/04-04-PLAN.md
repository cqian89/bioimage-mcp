def input_missing_error(
    message: str,
    missing_inputs: list[dict],
) -> StructuredError:
    """Create an INPUT_MISSING error with hints for each missing input.
    
    Args:
        message: Summary message
        missing_inputs: List of dicts with 'key', 'type', 'first_seen' info
    """
    details = [
        ErrorDetail(
            path=f"/inputs/{inp['key']}",
            expected=f"ArtifactRef ({inp['type']})",
            actual="not provided",
            hint=f"Provide a path or artifact ID for '{inp['key']}' (first used at step {inp['first_seen']['step_index']}, port {inp['first_seen']['port']})"
        )
        for inp in missing_inputs
    ]
    return StructuredError(
        code=INPUT_MISSING,
        message=message,
        details=details,
    )
```

def format_error_summary(error: StructuredError, context: dict | None = None) -> str:
    """Format a StructuredError into a human-readable summary.
    
    Args:
        error: The structured error
        context: Optional context like step_index, fn_id
        
    Returns:
        Multi-line human-readable string
    """
    lines = [f"Error: {error.message}"]
    lines.append(f"Code: {error.code}")
    
    if context:
        if "step_index" in context:
            lines.append(f"Step: {context['step_index']} ({context.get('fn_id', 'unknown')})")
    
    if error.details:
        lines.append("")
        lines.append("Details:")
        for i, detail in enumerate(error.details, 1):
            lines.append(f"  {i}. {detail.path}")
            if detail.expected:
                lines.append(f"     Expected: {detail.expected}")
            if detail.actual:
                lines.append(f"     Actual: {detail.actual}")
            if detail.hint:
                lines.append(f"     Hint: {detail.hint}")
    
    return "\n".join(lines)
```
